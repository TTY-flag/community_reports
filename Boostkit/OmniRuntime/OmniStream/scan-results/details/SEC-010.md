## 从输入验证视角的修复建议

### 类型安全访问

```cpp
void setColValue(int rowIndex, int colIndex, omnistream::VectorBatch* vectorBatch, nlohmann::json& node)
{
    auto& type = fieldTypes[colIndex];
    auto& name = fieldNames[colIndex];
    
    // 验证字段存在性
    if (!node.contains(name)) {
        vectorBatch->SetNullAt(colIndex, rowIndex);
        return;
    }
    
    auto& field = node[name];
    
    switch (type) {
        case omniruntime::type::DataTypeId::OMNI_INT:{
            // 验证类型匹配
            if (!field.is_number_integer()) {
                LOG_ERROR("Type mismatch at row " << rowIndex << " col " << colIndex);
                vectorBatch->SetNullAt(colIndex, rowIndex);
                return;
            }
            vectorBatch->SetValueAt(colIndex, rowIndex, field.get<int32_t>());
            break;
        }
        // ... 其他类型添加类似验证
    }
}
```
